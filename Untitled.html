<script>

    // JavaScript for Drag-and-Drop Functionality

    const scheduleList = document.getElementById('scheduleList');

    let draggedItem = null;


    // 1. 設置拖曳開始

    scheduleList.addEventListener('dragstart', (e) => {

        // 確保我們只處理排程項目

        if (e.target.classList.contains('schedule-item')) {

            draggedItem = e.target;

            setTimeout(() => {

                draggedItem.classList.add('dragging'); // 添加拖曳樣式

            }, 0);

        }

    });


    // 2. 設置拖曳結束

    scheduleList.addEventListener('dragend', (e) => {

        draggedItem.classList.remove('dragging');

        draggedItem = null;

    });


    // 3. 設置拖曳經過的區域 (避免預設行為)

    scheduleList.addEventListener('dragover', (e) => {

        e.preventDefault(); // 允許放下

    });


    // 4. 處理拖曳進入另一個項目上方

    scheduleList.addEventListener('dragenter', (e) => {

        e.preventDefault();

        const target = e.target.closest('.schedule-item');


        if (!target || target === draggedItem) return;


        // 判斷拖曳是發生在目標項目的上方還是下方

        const targetRect = target.getBoundingClientRect();

        const middleY = targetRect.top + targetRect.height / 2;


        if (e.clientY < middleY) {

            // 拖曳到上方：將 draggedItem 插入到 target 之前

            scheduleList.insertBefore(draggedItem, target);

        } else {

            // 拖曳到下方：將 draggedItem 插入到 target 之後

            scheduleList.insertBefore(draggedItem, target.nextSibling);

        }

    });


    // 5. 處理拖曳離開

    scheduleList.addEventListener('dragleave', (e) => {

        // 通常不需要特別處理

    });

</script>
